void BinaryInsertionDescendingSort (char *arr) 
{
	int i,k,middle,right,left; // два счетчика,индекс среднего элемента, индекс правой и левой границ
	char temp;					//переменная для хранения значений при перемещении элементов
	for (i=1;i<length;i++)  //каждый из элементов массива, начиная со второго фактического
	{
		left=0;             //определеям начальную левую и правую границы
		right=i;		
		while (right-left>0)  //пока границы не станут совпадать, ищем центральный элемент в отсортированной части
		{
		middle=(left+right)/2;  
		if (arr[i]<arr[middle]) left=middle+1;  // если обрабатываемый элемент массива меньше или больше центрального, то сдвигаем границы
		else 
		 {
		 	if (arr[i]==arr[middle]) 
			 {
				 left=middle+1;
				 right=left;           //если наш элемент совпадает с центральным, то прекращаем поиски, приравнивая границы
			 }
		 	else 
			 {
			 	right=middle;
			 }
			 
		 }	
		}
		
		temp=arr[i];           //сдвигаем все элементы после найденной позиции вправо, что бы освободить ячейку
		for (k=i;k>left;k--)
		{
			arr[k]=arr[k-1];
		}
		arr[middle+1]=temp;  //ставим наш элемент на его новое место
	
	}
}
